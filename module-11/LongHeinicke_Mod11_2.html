<!DOCTYPE html>

<html lang="en">

<head>

    <title>CSD 340 Web Development with HTML and CSS</title>

    <script>

        let itemArray = [];


        function addElement() {

            const input = document.getElementById('textEntry').value.trim();

            if (input) {

                itemArray.push(input);

                alert(`Added "${input}" at position ${itemArray.length}`);

                document.getElementById('textEntry').value = '';

            }

        }


        function deleteLastEntry() {

            if (itemArray.length > 0) {

                const removedItem = itemArray.pop();

                alert(`Removed "${removedItem}"`);

            } else {

                alert('Array is already empty!');

            }

        }


        function displayArray() {

            let output = '';

            if (itemArray.length > 0) {

                for (let i = 0; i < itemArray.length; i++) {

                    output += `${i + 1}. ${itemArray[i]}\n`;

                }

            } else {

                output = 'The array is currently empty';

            }

            alert(output);

        }

    </script>

</head>


<body>

    <h1>Assignment 11.2: JavaScript - Part 3</h1>

    

    <div>

        <input type="text" id="textEntry" placeholder="Enter text here">

        <button onclick="addElement()">Add</button>

        <button onclick="deleteLastEntry()">Delete Last Entry</button>

        <button onclick="displayArray()">Display</button>

    </div>

</body>

</html>
