    <script>

        // Original book array

        const books = [

            "Lord Of the Rings",

            "Harry Potter",

            "Legends and Lattes",

            "Pride and Prejudice",

            "Babel",

            "The Poppy War",

            "Godkiller",

            "The Divine Comedy",

            "Howl’s Moving Castle",

            "The Dresden Files"

        ];


        // Initial display

        displayBooks(books);


        function sortAscending() {

            const sorted = [...books].sort((a, b) => a.localeCompare(b));

            displayBooks(sorted);

        }


        function sortDescending() {

            const sorted = [...books].sort((a, b) => b.localeCompare(a));

            displayBooks(sorted);

        }


        function displayBooks(bookArray) {

            const bookList = document.getElementById('bookList');

            bookList.innerHTML = '<h3>Top 10 Favorite Books</h3><ol>' + 

                bookArray.map(book => `<li>${book}</li>`).join('') + 

                '</ol>';

        }

    </script>
